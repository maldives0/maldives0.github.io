//sol1
function solution(a, b) {
    var answer = 0;
    let ab = [];
    let sol = [a, b];
    if (a == b) {
        return a;
    } else {
        let c = sol.sort((c, p) => c - p);

        for (let i = 0; i <= c[1]; i++) {
            if (i >= c[0]) {
                ab.push(parseInt(i));
            }
        }
        ab.map((v, i) => {
            return answer += ab[i];
        });
    }
    return answer;
}
